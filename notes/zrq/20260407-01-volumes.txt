    DockerVolumeMountStorage
        validate()
            check space from volume pool
            later - booking space from the pool

        prepare()
            create docker-volume

        release()
            delete docker-volume

    DockerBindMountStorage
        validate()
            check path is accessible
            if endsWith("/")
                checkDirectory(mode)
            else
                checkFile(mode)

        checkFile(mode)
            if file exists
                check mode matches
            else
                check parent writable
            if mode = writeable
                check available space

        checkDirectory(mode)
            if directory exists
                check mode
            else
                check parent writable

        release()
            --

    DockerSimpleCompute

        prepare()
            for each volume
                MountList.add
                    DockerHttpDataResource.getStorage()
                        => DockerVolumeMountStorage implements DockerVolumeMount
                        DockerVolumeMountStorage.getMountIdent()
                            => docker-volume.getVolumeName()
                            => "{volume-ident}:{mount-path} ro/rw"

                    DockerFileDataResource.getStorage()
                        => DockerBindMountStorage implements DockerVolumeMount
                        DockerBindMountStorage.getMountIdent()
                            => getHostPath()
                            => "{host-path}:{mount-path} ro/rw"


        release()
            Stop container
            Delete container
