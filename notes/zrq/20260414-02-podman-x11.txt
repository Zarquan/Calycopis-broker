# Bind mount the X11 socket to run graphical program in a container.
#[user@desktop]

    #
    # Thanks to Major Hayden :-)
    # https://major.io/p/run-xorg-applications-with-podman/

    podman run \
        --rm \
        --tty \
        --interactive \
        --env "DISPLAY=${DISPLAY}" \
        --volume "/tmp/.X11-unix:/tmp/.X11-unix:rw" \
        --security-opt "label=type:container_runtime_t" \
        ghcr.io/ivoa/calycopis/developer-tools:2026.04.12 \
            bash

            dnf install -y xeyes
            xeyes


    #
    # The full form

    podman run \
        --rm \
        --tty \
        --interactive \
        --pod  calycopis-pod \
        --name calycopis-dev \
        --volume "${HOME}/.m2:/root/.m2:rw,Z" \
        --volume "${HOME}/.cursor:/root/.cursor:rw,Z" \
        --volume "${CALYCOPIS_ROOT}:/calycopis:rw,Z" \
        --volume "${HELIOPHORUS_ROOT}:/heliophorus:rw,Z" \
        --env "SSH_AUTH_SOCK=/tmp/ssh-sock" \
        --volume "${SSH_AUTH_SOCK}:/tmp/ssh-sock:rw,Z" \
        --env "CONTAINER_HOST=unix:///run/podman/podman.sock" \
        --volume "${XDG_RUNTIME_DIR}/podman/podman.sock:/run/podman/podman.sock:rw,Z" \
        --env "DISPLAY=${DISPLAY}" \
        --volume "/tmp/.X11-unix:/tmp/.X11-unix:rw" \
        --security-opt "label=type:container_runtime_t" \
        ghcr.io/ivoa/calycopis/developer-tools:2026.04.12 \
            bash

            dnf install -y xeyes
            xeyes

    #
    # All three tunnels work.
    # SSH, Docker, and X11.
    #
