# Create some offers.
#[root@spring-builder]

    createrequest=$(
        mktemp --suffix '.yaml'
        )
    createresponse=$(
        mktemp --suffix '.yaml'
        )

cat > "${createrequest}"  << EOF
executable:
  properties:
  name: Test executable
  type: urn:jupyter-notebook-0.1
  notebook: http://there
schedule:
  requested:
  - duration:
      min: PT2H
resources:
  compute:
    - name: Compute 001
      type: urn:simple-compute-resource
      cores:
        min: 8
      memory:
        min: 8
EOF

    curl \
        --silent \
        --show-error \
        --location \
        --header "Content-Type: application/yaml" \
        --data-binary @"${createrequest}" \
        --header "Accept: application/yaml" \
        "http://127.0.0.1:8082/offerset" \
    | tee "${createresponse}" \
    | yq '.offers[].schedule'

        updatehref=$(
            yq '.offers[2].href' "${createresponse}"
            )

        # Accept 1 (reject 3)
        curl \
            --silent \
            --show-error \
            --location \
            --header "Content-Type: application/yaml" \
            --data-binary @"${updaterequest}" \
            --header "Accept: application/yaml" \
            "${updatehref}" \
        | tee "${updateresponse}" \
        | yq '{"state":.state, "schedule":.schedule}'

    done
